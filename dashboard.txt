<?php
// dashboard.php (saved as dashboard.txt)
require_once 'config.php';

// Fetch recent posts
$stmt = $pdo->prepare("SELECT * FROM posts ORDER BY created_at DESC LIMIT 10");
$stmt->execute();
$recent_posts = $stmt->fetchAll();

// Fetch connected accounts
$stmt = $pdo->prepare("SELECT * FROM social_accounts WHERE user_id = ?");
$stmt->execute([1]); // Mock user ID
$accounts = $stmt->fetchAll();

header('Content-Type: application/json');
echo json_encode([
    'posts' => $recent_posts,
    'accounts' => $accounts
]);
?>
